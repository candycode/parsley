45 class RefParser : public IParser {
46 public:
47  typedef IParser* IParserPtr; //smart pointers ?
48  RefParser() : ref_( 0 ), memStartPos_( 0 ), memEndPos_( 0 ) {}
49  RefParser( IParser& ip ) : ref_( &ip ), memStartPos_( 0 ),
50  memEndPos_( 0 ) {}
51  const Values& GetValues() const { return values_; }
52  const ValueType& operator[]( const KeyType& k ) const {
53  return ref_->operator[]( k );
54  }
55  bool Parse( InStream& is ) {
56  assert( ref_ && "NULL PARSER REFERENCE" );
57  // little memoization logic: if text already parsed simply skip to
58  // end of parsed text
59  if( memStartPos_ == is.tellg() && !(ref_->GetValues().empty()) ) {
60  is.seekg( memEndPos_ );
61  return true;
62  }
63  const StreamPos p = is.tellg();
64  const bool ok = ref_->Parse( is );
65  // if parsed record begin and end of parsed text
66  if( ok ) {
67  memStartPos_ = p;
68  memEndPos_ = is.tellg();
69  values_ = ref_->GetValues();
70  }
71  return ok;
72  }
73  RefParser* Clone() const { return new RefParser( *this ); }
74 private:
75  StreamPos memStartPos_;
76  StreamPos memEndPos_;
77  mutable Values values_;
78  IParserPtr ref_;
79 };
