28  bool Parse( InStream& is ) {
29  assert( ref_ && "NULL PARSER REFERENCE" );
30  // little memoization logic: if text already parsed simply skip to
31  // end of parsed text
32  if( memStartPos_ == is.tellg() && !(ref_->GetValues().empty()) ) {
33  is.seekg( memEndPos_ );
34  return true;
35  }
36  const StreamPos p = is.tellg();
37  const bool ok = ref_->Parse( is );
38  // if parsed record begin and end of parsed text

40  memStartPos_ = p;
41  memEndPos_ = is.tellg();
42  values_ = ref_->GetValues();
43  }
44  return ok;
45  }
46  RefParser* Clone() const { return new RefParser( *this ); }
47 private:
48  StreamPos memStartPos_;
