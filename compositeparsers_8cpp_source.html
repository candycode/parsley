54 // TEST COMPOSITE PARSERS
55 void TestCompositeParsers() {
56  std::cout << "\nCOMPOSITE PARSERS TEST" << std::endl;
57  std::cout << "========================" << std::endl;
58  int count = 0;
59  //1 AND lexer
60  std::istringstream iss( "c 1 6 0.4 -2.3 1.1" );
61  InStream is( iss );
62  AndParser al( A("n") > U("#") > U("e") > F("x") > F("y") > F("z") );
63  assert( al.Parse( is ) );
64  assert( al[ "n" ] == String( "c" ) );
65  assert( al[ "#" ] == unsigned( 1 ) );
66  assert( al[ "e" ] == unsigned( 6 ) );
67  assert( al[ "x" ] == 0.4 );
68  assert( al[ "y" ] == -2.3 );
69  assert( al[ "z" ] == 1.1 );
70  std::cout << ++count << ' ';
71  //2 TUPLE
72  std::istringstream iss2( " 2.1 2.2 2.3 " );
73  InStream is2( iss2 );
74  TupleParser< 3 > tl( FloatParser(), "coord" );
75  assert( tl.Parse( is2 ) );
76  std::vector< Any > v = tl[ "coord" ];
77  assert( v[ 0 ] == 2.1 && v[ 1 ] == 2.2 && v[ 2 ] == 2.3 );
78  std::cout << ++count << ' ';
79  //3 TUPLE
80  std::istringstream iss3( " 2.1 2.2 2.3 4" );
81  InStream is3( iss3 );
82  TupleParser<> tl2( FloatParser(), "coord" );
83  assert( tl2.Parse( is3 ) );
84  std::vector< Any > v2 = tl2[ "coord" ];
85  assert( v2[ 0 ] == 2.1 && v2[ 1 ] == 2.2
86  && v2[ 2 ] == 2.3 && v2[ 3 ] == 4.0 );
87  std::cout << ++count << ' ';
88  //4 TUPLE
89  std::istringstream iss4( " string1 | string2 | string3 | string4" );
90  InStream is4( iss4 );
91  Parser tl3( *( ( AL(DONT_SKIP_BLANKS) >= __ >= A("") >= __ >= C("|") )
92  /*OR*/ / ( AL(DONT_SKIP_BLANKS) >= __ >= A("") >= eof_ )
93  ) );
94  assert( tl3.Parse( is4 ) );
95  std::list< Any > vs = tl3[ "" ];
96  assert( *(vs.begin()) == S("string1")
97  && *(++vs.begin()) == S("string2")
98  && *(++++vs.begin() ) == S("string3")
99  && *( vs.rbegin() ) == S("string4" ) );
100  std::cout << ++count << ' ';
101  //5 TUPLE
102  std::istringstream
103  iss5( "| string 1 | string 2 | string 3 | string 4 | gfdgfd" );
104  InStream is5( iss5 );
105  AndParser separator(DONT_SKIP_BLANKS );
106  separator >= __ >= C( "|" ) ;
107  TupleParser<> tl4( NotParser<OrParser>( separator / NextLineParser() ),
108  "strings", separator > __, separator,
109  PassThruParser(), SKIP_BLANKS );
110  assert( tl4.Parse( is5 ) );
111  std::vector< Any > vs2 = tl4[ "strings" ];
112  assert( vs2.size() == 5 );
113  //using a pass-through as the end lexer means we always get something added
114  //as the last element
115  assert( vs2[ 0 ] == S("string 1") );
116  assert( vs2[ 1 ] == S("string 2") );
117  assert( vs2[ 2 ] == S("string 3") );
118  assert( vs2[ 3 ] == S("string 4") );
119  std::cout << ++count << ' ';
120  //6 OR lexer
121  std::istringstream oriss1( "c 1 6 0.4 -2.3 1.1" );
122  std::istringstream oriss2( " 2.1 2.2 2.3 " );
123  InStream oris1( oriss1 );
124  InStream oris2( oriss2 );
125  OrParser ol = al / tl;
126  assert( ol.Parse( oris1 ) );
127  assert( ol[ "n" ] == String( "c" ) );
128  assert( ol[ "#" ] == unsigned( 1 ) );
129  assert( ol[ "e" ] == unsigned( 6 ) );
130  assert( ol[ "x" ] == 0.4 );
131  assert( ol[ "y" ] == -2.3 );
132  assert( ol[ "z" ] == 1.1 );
133  assert( ol.Parse( oris2 ) );
134  std::vector< Any > v4 = ol[ "coord" ];
135  assert( v4[ 0 ] == 2.1 && v4[ 1 ] == 2.2 && v4[ 2 ] == 2.3 );
136  std::cout << ++count << ' ';
137  std::cout << std::endl;
138 }
