360 class FloatParser : public IParser {
361 public:
362  typedef Values::value_type::second_type ValueType;
363  typedef Values::key_type KeyType;
364  FloatParser( const FloatParser& p )
365  : name_( p.name_ ), valueMap_( p.valueMap_ ) {}
366  FloatParser( const KeyType& name = KeyType() ) : name_( name ) {}
367  virtual const String& GetText() const { return token_ ; }
371  virtual bool Parse( InStream& is ) {
372  token_.clear();
373  valueMap_.clear();
374  if( !is.good() ) return false;
375  // in case support for locale-dependent decimal separator is needed:
376  //Char decimalSeparator_ =
377  // std::use_facet< std::numpunct< Char > >( is.getloc() )
378  // .decimal_point ();
379  bool ok = false; {
380  REWIND r( ok, is );
381  const Char c = is.get();
382  if( !is.good() ) return false;
383  if( c == '+' || c == '-' ) {
384  if( Apply( &FloatParser::MatchUnsigned, is, c ) ) {
385  token_ = c + token_;
386  }
387  }
388  else if( c == '.' ) {
389  if( Apply( &FloatParser::MatchFractional, is, c ) ) {
390  token_ = c + token_;
391  }
392  }
393  else if( parsley::IsDigit( c ) != 0 ) {
394  is.unget();
395  Apply( &FloatParser::MatchUnsigned, is, 0 );
396  }
397  else return false;
398  ok = token_.length() > 0;
399  }
400  return ok;
401  }
406  const Values& GetValues() const {
407  if( valueMap_.empty() && !token_.empty() ) {
408  valueMap_.insert(

411  }
412  return valueMap_;
413  }

430  typedef bool( FloatParser::*MatchMethod )( InStream& );

432  bool Apply( MatchMethod f, InStream& is, Char last ) {
433  if( !is.good() ) return false;
434  const String tmp = token_; // save current value
435  const StreamPos pos = is.tellg();
436  const bool m = (this->*f)( is );
437  if( !m ) {
438  //restore previous value
439  token_ = tmp;
440  //restore stream pointer
441  is.seekg( pos );
442  }
443  //last == 0 is a signal that enough input has been validated to make
444  //the token a valid float
445  return last == 0 || m;
446  }

448  bool MatchUnsigned( InStream& is ) {
449  if( !is.good() ) return false;
450  const Char c = is.get();
451  if( !is.good() ) return false;
452  if( parsley::IsDigit( c ) != 0 ) {
453  token_.push_back( c );
454  Apply( &FloatParser::MatchUnsigned, is, 0 );
455  return true;
456  }
457  else if( c == '.' ) {
458  token_.push_back( c );
459  return Apply( &FloatParser::MatchFractional, is, c );
460  }
461  is.unget();
462  return false;
463  }

465  bool MatchFractional( InStream& is ) {
466  if( !is.good() ) return false;
467  const Char c = is.get();
468  if( !is.good() ) return false;
469  if( parsley::IsDigit( c ) ) {
470  token_.push_back( c );
471  Apply( &FloatParser::MatchFractional, is, 0 );
472  return true;
473  }
474  else if( c == 'E' || c == 'D' ||
475  c == 'e' || c == 'd' ) {
476  // (as of 2009) problem: on windows 'D' (with both vc++ and migw)
477  // is properly understood but not on linux/mac: need to force a
478  // 'D' -> 'E' translation
479  token_.push_back( 'E' );
480  return Apply( &FloatParser::MatchExponent, is, c );
481  }
482  is.unget();
483  return false;
484  }

486  if( !is.good() ) return false;

488  if( !is.good() ) return false;

490  token_.push_back( c );

492  }
493  else if( parsley::IsDigit( c ) ) {
494  token_.push_back( c );

496  return true;

773 class TupleParser : public IParser {
774 public:
775  typedef Values::value_type::second_type ValueType;
776  typedef Values::key_type KeyType;
777  typedef InStream::char_type Char;
778  typedef std::list< Parser > Parsers;
779  TupleParser() {}
789  TupleParser( const Parser& v, const ValueID& name = ValueID(),
790  const Parser& b = BlankParser(),
791  const Parser& s = BlankParser(),
792  const Parser& e = BlankParser(), bool blanks = false )
793  : skipBlanks_( blanks ),
794  beginParser_( b ), valueParser_( v ),
795  separatorParser_( s ), endParser_( e ), name_( name ) {}
796  TupleParser( const TupleParser& l )
797  : skipBlanks_( l.skipBlanks_ ), name_( l.name_ ),
798  beginParser_( l.beginParser_ ),
799  valueParser_( l.valueParser_ ),
800  separatorParser_( l.separatorParser_ ),
801  endParser_( l.endParser_ ), values_( l.values_ ),
802  valueMap_( l.valueMap_ )
803  {}
809  bool Parse( InStream& is ) {
810  values_.clear();
811  valueMap_.clear();
812  if( !is.good() ) return false;
813  int counter = 0;
814  bool ok = false; {
815  // record current file pointer and rewind when exiting scope in case
816  // parser not applied; nested scope required if local variable used
817  // as signal
818  REWIND r( ok, is );
819  // apply begin parser and return false if it fails
820  if( skipBlanks_ ) SkipBlanks( is );
821  if( !beginParser_.Parse( is ) ) return false;
822  bool endReached = false;
823  std::vector< ValueType >::size_type s = 0;
824  // start applying value parser
825  if( skipBlanks_ ) SkipBlanks( is );
826  while( valueParser_.Parse( is ) ) {
827  // parsed values: add values to value array
828  AddValue( valueParser_.GetValues() );
829  // increment value counter
830  ++counter;
831  // apply separator parser and exit loop if it fails
832  if( !separatorParser_.Parse( is ) ) break;
833  // value & separator parser applied, if required skip blanks
834  // before next value or end separator
835  if( skipBlanks_ ) SkipBlanks( is );
836  }
837  // exited from loop: either end separator or wrong expression found
838 
839  // apply end separator: if it succeeds it means the end separator was
840  // parsed successfully and the end of the tuple was reached
841  if( skipBlanks_ ) SkipBlanks( is );
842  endReached = endParser_.Parse( is );
843 
844  // if end reached and number of parsed values is correct return true,
845  // false otherwhise
846  ok = endReached && ( counter == Size()
847  || ( counter > 0 && Size() < 0 ) );
848  }
849  return ok;
850  }

855  if( valueMap_.empty() && !values_.empty() ) {
856  valueMap_.insert( std::make_pair( name_, values_ ) );
857  }

859  }
