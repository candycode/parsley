430  bool Apply( MatchMethod f, InStream& is, Char last ) {
431  if( !is.good() ) return false;

824  // increment value counter
825  ++counter;
826  // apply separator parser and exit loop if it fails

828  // value & separator parser applied, if required skip blanks
829  // before next value or end separator
830  if( skipBlanks_ ) SkipBlanks( is );
831  }
832  // exited from loop: either end separator or wrong expression found
833 
834  // apply end separator: if it succeeds it means the end separator was
835  // parsed successfully and the end of the tuple was reached
836  if( skipBlanks_ ) SkipBlanks( is );
837  endReached = endParser_.Parse( is );
