87  bool HandleValues( StateID sid, const Values& v ) {
88  Program::iterator levelBegin, levelEnd;
89  switch( sid ) {
90  case EXPRESSION_BEGIN:
91  state_->program_.push_back( OP( LEFT_P ) );
92  levelStack_.push( --( state_->program_.end() ) );
93  break;
94  case EXPRESSION_END: {
95  state_->program_.push_back( OP( RIGHT_P ) );
96  levelEnd = --( state_->program_.end() );
97  levelBegin = levelStack_.top();
98  levelStack_.pop();
99  const double V = state_->Evaluate( ++levelBegin, levelEnd );
100  state_->program_.erase( --levelBegin, ++levelEnd );
101  state_->program_.push_back( OP( VAL, V ) );
102  break;
103  }
104  case VALUE: state_->program_.push_back(
105  OP( VAL, v.find("operand")->second ) ) ;
106  break;
107  case OPERATOR: {
108  const String op = v.find( "operator" )->second;
109  if( op == "+" ) state_->program_.push_back( OP( ADD ) );
110  else if( op == "-" ) state_->program_.push_back( OP( SUB ) );
111  else if( op == "*" ) state_->program_.push_back( OP( MUL ) );
112  else if( op == "/" ) state_->program_.push_back( OP( DIV ) );
113  }
114  break;
115  case EOF_: if( !levelStack_.empty() ) {
116  throw std::logic_error( "Unmatched parenthesis" );
117  return false; //in case no exception handling enabled
118  }
119  std::cout << state_->Evaluate(
120  state_->program_.begin(), state_->program_.end() );
121  break;
122  default: break;
123  }
124  Values::const_iterator i;
125  for( i = v.begin(); i != v.end(); ++i ) {
126  std::cout << '(' << i->first << " = " << i->second << ')' << ' ';
127  }
128  std::cout << '\n';
129  return true;
130  }

136  struct State {
137  Program program_;
138  double Evaluate( Program::iterator b, Program::iterator e ) {
139  // termination condition
140  Program::iterator ei = e;
141  if( b == --ei ) return b->value_;
142  Program::iterator i = std::find( b, e, OP( ADD ) );
143  Program::iterator l;
144  Program::iterator r;
145  if( i != e ) {
146  l = i;
147  r = i;
148  if( i == b ) return 0 + Evaluate( ++r, e );
149  else return Evaluate( b, l ) + Evaluate( ++r, e );
150  }
151  i = std::find( b, e, OP( SUB ) );
152  if( i != e ) {
153  l = i;
154  r = i;
155  if( i == b ) return 0 - Evaluate( ++r, e );
156  else return Evaluate( b, l ) - Evaluate( ++r, e );
157  }
158  i = std::find( b, e, OP( MUL ) );
159  if( i != e ) {
160  l = i;
161  r = i;
162  return Evaluate( b, l ) * Evaluate( ++r, e );
163  }
164  i = std::find( b, e, OP( DIV ) );
165  if( i != e ) {
166  l = i;
167  r = i;
168  // could catch division by zero exception
169  return Evaluate( b, l ) / Evaluate( ++r, e );
170  }
171  throw std::logic_error( "Invalid expression" );
172  }

182 void TestExprEval() {
183  std::cout << "EXPRESSION EVALUATOR" << std::endl;

186  const String EXPR = "1+3*(4-(2-1/3.3))";
187  std::cout << "Expression: " << EXPR << " = " << 1+3*(4-(2-1/3.3))
188  << std::endl;

190  InStream is( iss );
191  SmartPtr< eval::Print::State > state( new eval::Print::State );
192  StateManager PRINT( ( eval::Print( state ) ) );
193  ParserManager pm( PRINT );
194  pm.SetBeginEndStates( eval::START, eval::EOF_ );
195  pm.AddState( eval::START, eval::VALUE );
196  pm.AddState( eval::START, eval::EXPRESSION_BEGIN );
197  pm.AddState( eval::EXPRESSION_BEGIN, eval::EXPRESSION_BEGIN );
198  pm.AddState( eval::EXPRESSION_BEGIN, eval::VALUE );
199  pm.AddState( eval::VALUE, eval::EXPRESSION_END );
200  pm.AddState( eval::EXPRESSION_END, eval::EXPRESSION_END );
201  pm.AddState( eval::EXPRESSION_END, eval::EOF_ );
202  pm.AddState( eval::EXPRESSION_END, eval::OPERATOR );
203  pm.AddState( eval::VALUE, eval::OPERATOR )
204  .AddState( eval::VALUE, eval::EOF_ );
205  pm.AddState( eval::OPERATOR, eval::EXPRESSION_BEGIN );
206  pm.AddState( eval::OPERATOR, eval::VALUE );
207 
208  pm.SetParser( eval::VALUE,
209  ( AL( DONT_SKIP_BLANKS ) >= __ >= F("operand") ) );
210  pm.SetParser( eval::OPERATOR,
211  ( AL( DONT_SKIP_BLANKS ) >= __ >=
212  ( C("+","operator") /
213  C("-","operator") /
214  C("/","operator") /
215  C("*","operator") ) ) );
216  pm.SetParser( eval::EXPRESSION_BEGIN, ( __ > C("(","open") ) );
217  pm.SetParser( eval::EXPRESSION_END, ( __ > C(")","closed") ) );
218  pm.SetParser( eval::EOF_, ( eof_ ) );
219 
220  pm.Apply( is, eval::START );
221 }
