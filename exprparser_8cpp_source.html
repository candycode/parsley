75  break;
76  }

112  // termination condition
113  Program::iterator ei = e;
114  if( b == --ei ) return b->value_;
115  Program::iterator i = std::find( b, e, OP( ADD ) );
116  Program::iterator l;
117  Program::iterator r;
118  if( i != e ) {
119  l = i;
120  r = i;
121  if( i == b ) return 0 + Evaluate( ++r, e );
122  else return Evaluate( b, l ) + Evaluate( ++r, e );
123  }
124  i = std::find( b, e, OP( SUB ) );
125  if( i != e ) {
