83  char_type get() {
84  assert( isp_ != 0 && "NULL STREAM POINTER" );
85  char_type c = 0;
86  if( isp_->good() ) {
87  c = isp_->get();
88  } else {
89  throw std::logic_error( "Attempt to read from invalid stream" );
90  return c; // in case exception handling disabled
91  }
92  if( c == EOL_ ) {
93  ++lines_;
94  lineChars_ = 0;
95  eols_.push_back( streamoff( isp_->tellg() ) - 1 );
96  }
97  else ++lineChars_;
98  // this is to ensure that the eol list doesn't get too big, could
99  // use a custom fixed size circular buffer
100  if( eols_.size() > RESIZE_THRESHOLD ) {
101  eols_.erase( eols_.begin(), eols_.begin() + RESIZE_THRESHOLD / 2 );
102  }
103  return c;
104  }

195  void BackwardSeek( streampos p ) {
196  assert( isp_ != 0 && "NULL STREAM POINTER" );
197  isp_->seekg( p );
198  const streamoff sp = p;
199  reolIt_ = std::find_if( eols_.rbegin(), eols_.rend(),
200  std::bind1st( std::greater< streamoff >(), sp ) );
